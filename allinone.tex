\documentclass{standalone}

\usepackage[latin1]{inputenc}
\usepackage{tikz}
\usetikzlibrary{shapes,arrows}
\begin{document}
\pagestyle{empty}


% Define block styles
\tikzstyle{decision} = [diamond, draw, fill=blue!3, 
    text width=6em, text badly centered, node distance=3cm, inner sep=0pt]
\tikzstyle{block} = [rectangle, draw, fill=green!3, 
    text width=6em, text centered, rounded corners, minimum height=3.5em]
\tikzstyle{line} = [draw, very thick, -latex']
\tikzstyle{cloud} = [draw, ellipse,fill=red!3, node distance=2cm,
    minimum height=2em]
    
\begin{tikzpicture}[node distance =3cm, auto]
    % Place nodes
    \node [cloud] (start) {start};
    \node [block, below of=start] (listbens) {peruse list of beneficiaries};
    \node [decision, below of=listbens] (inlist) {is needed beneficiary in list?};
    \node [block, left of=inlist, xshift=-.5cm] (addbentolist) {add beneficiary to list};
    \node [block, below of=inlist] (recordben) {record beneficiary};
    \node [decision, below of=recordben] (benneedrecorded) {record another beneficiary?};
    \node [block, below of=benneedrecorded] (selectben) {select recorded beneficiary};
    \node [block, below of=selectben] (attrlist) {view default list of attributes};
    \node [decision, below of=attrlist] (attrlistedit) {edit default list of attributes?};
    \node [block, left of=attrlistedit, xshift=-.5cm] (addrmattr) {add or remove attributes};
    \node [block, below of=attrlistedit] (judge) {judge site};
    \node [block, below of=judge] (explain) {explain judgment};
    \node [decision, below of=explain] (anotherben) {another beneficiary?};
    \node [cloud, below of=anotherben, yshift=-.5cm] (done) {done};
    % Draw edges
    \path [line] (start) -- (listbens);
    \path [line] (listbens) -- (inlist);
    \path [line] (inlist) -- node {no}(addbentolist);
    \path [line] (addbentolist) -- (listbens);
    \path [line] (inlist) -- node {yes}(recordben);
    \path [line] (recordben) -- (benneedrecorded);
    \path [line] (benneedrecorded.west) --++ (-4cm,0cm) |- (listbens);
    \path [line] (benneedrecorded) -- node {no} (selectben);
    \path [line] (selectben) -- (attrlist);
    \path [line] (attrlist) -- (attrlistedit);
    \path [line] (attrlistedit) -- node {yes} (addrmattr);
    \path [line] (addrmattr) |- (judge);
    \path [line] (attrlistedit) -- (judge);
    \path [line] (judge) -- (explain);
    \path [line] (explain) -- (anotherben);
    \path [line] (anotherben.west)  --++ (-4cm,0cm) |- node [near start] {yes} (selectben);
    \path [line] (anotherben) -- node {no} (done);
\end{tikzpicture}

\end{document}
